// Zoox Page Architecture Analyzer - FIXED VERSION
(function() {
    class ZooxArchitectureAnalyzer {
        constructor() {
            this.container = document.querySelector('.PageHome_contentBespoke__ZkpR3');
            this.framework = this.detectFramework();
            this.animationLibrary = this.detectAnimationLibrary();
            this.cssArchitecture = this.analyzeCSSArchitecture();
            this.htmlStructure = this.analyzeHTMLStructure();
            this.javascriptControls = this.analyzeJavaScriptControls();
            this.buildTools = this.detectBuildTools();
        }

        detectFramework() {
            const frameworks = {
                react: !!(window.React || document.querySelector('[data-reactroot]')),
                nextjs: !!(window.__NEXT_DATA__),
                vue: !!(window.Vue),
                angular: !!(window.ng),
                gsap: !!(window.gsap),
                jquery: !!(window.jQuery)
            };

            // Check for Next.js specifically
            if (window.__NEXT_DATA__) {
                return {
                    name: 'Next.js',
                    version: window.__NEXT_DATA__.buildId ? 'with App Router' : 'Pages Router',
                    data: window.__NEXT_DATA__
                };
            }

            // Check for React
            if (frameworks.react) {
                return {
                    name: 'React',
                    version: this.detectReactVersion()
                };
            }

            return frameworks;
        }

        detectReactVersion() {
            if (document.querySelector('[data-reactroot]')) {
                return 'React 16+';
            }
            if (document.querySelector('[data-reactid]')) {
                return 'React 0.14-15';
            }
            return 'Unknown React version';
        }

        detectAnimationLibrary() {
            const animations = {
                gsap: !!(window.gsap),
                scrollTrigger: !!(window.ScrollTrigger),
                framer: !!document.querySelector('[data-framer]'),
                aos: !!document.querySelector('[data-aos]'),
                lottie: !!window.lottie,
                three: !!window.THREE,
                cssAnimations: this.detectCSSAnimations()
            };

            if (animations.gsap) {
                return {
                    library: 'GSAP',
                    version: window.gsap.version || 'Unknown',
                    plugins: this.detectGSAPPlugins(),
                    instances: this.getGSAPInstances()
                };
            }

            return animations;
        }

        detectGSAPPlugins() {
            const plugins = [];
            
            if (window.ScrollTrigger) plugins.push('ScrollTrigger');
            if (window.Draggable) plugins.push('Draggable');
            if (window.SplitText) plugins.push('SplitText');
            if (window.ScrollToPlugin) plugins.push('ScrollToPlugin');
            
            return plugins;
        }

        getGSAPInstances() {
            const instances = [];
            
            // Look for GSAP-related attributes
            const animatedElements = document.querySelectorAll('[style*="transform"], [style*="opacity"]');
            animatedElements.forEach(el => {
                const styles = window.getComputedStyle(el);
                if (styles.transform !== 'none' || styles.opacity !== '1') {
                    instances.push({
                        element: el.className.slice(0, 50) || el.tagName,
                        transform: styles.transform,
                        opacity: styles.opacity
                    });
                }
            });
            
            return instances;
        }

        detectCSSAnimations() {
            const animations = [];
            const styleSheets = document.styleSheets;
            
            for (let sheet of styleSheets) {
                try {
                    const rules = sheet.cssRules || sheet.rules;
                    if (!rules) continue;
                    
                    for (let rule of rules) {
                        // Check for keyframes
                        if (rule.type === CSSRule.KEYFRAMES_RULE) {
                            animations.push({
                                name: rule.name,
                                type: 'keyframes'
                            });
                        }
                        
                        // Check for animations in style rules
                        if (rule.style && rule.style.animationName) {
                            animations.push({
                                selector: rule.selectorText,
                                animation: rule.style.animation
                            });
                        }
                    }
                } catch (e) {
                    continue;
                }
            }
            
            return animations;
        }

        analyzeCSSArchitecture() {
            return {
                methodology: this.detectCSSMethodology(),
                components: this.extractComponents(),
                cssModules: this.detectCSSModules(),
                stylingApproach: this.detectStylingApproach(),
                responsiveBreakpoints: this.getResponsiveBreakpoints(),
                cssVariables: this.getCSSVariables(),
                importantSelectors: this.getImportantSelectors()
            };
        }

        detectCSSMethodology() {
            const classes = this.container?.className || '';
            const methodologies = {
                bem: classes.includes('__') && classes.includes('--'),
                suit: classes.includes('-') && classes.includes('--'),
                atomic: classes.match(/[a-z][A-Z]/g), // CamelCase classes
                tailwind: classes.includes('flex') || classes.includes('p-') || classes.includes('m-'),
                styledComponents: !!document.querySelector('[class^="sc-"]')
            };
            
            // Zoox specifically uses BEM-like naming (HomepageContentBespoke_root__W9khF)
            if (classes.includes('__') || classes.match(/[A-Z][a-z]+[A-Z]/)) {
                return 'BEM-like / CSS Modules';
            }
            
            return methodologies;
        }

        detectCSSModules() {
            // Check for CSS Modules pattern (filename_component__hash)
            const classes = this.container?.className || '';
            const matches = classes.match(/[a-zA-Z]+_[a-zA-Z]+__[a-zA-Z0-9]+/);
            
            if (matches) {
                return {
                    pattern: matches[0],
                    explanation: 'CSS Modules with [filename]_[className]__[hash] pattern'
                };
            }
            
            return null;
        }

        extractComponents() {
            const components = {};
            
            // Extract component names from class names
            const allElements = document.querySelectorAll('[class*="Homepage"], [class*="Content"], [class*="Bespoke"]');
            
            allElements.forEach(el => {
                const classes = el.className.split(' ');
                classes.forEach(cls => {
                    // Match component names like HomepageContentBespoke, TextContent, etc.
                    const matches = cls.match(/[A-Z][a-z]+[A-Z][a-zA-Z]+/g);
                    if (matches) {
                        matches.forEach(match => {
                            components[match] = (components[match] || 0) + 1;
                        });
                    }
                });
            });
            
            return components;
        }

        detectStylingApproach() {
            // Check for different styling approaches
            const styles = {
                inlineStyles: !!document.querySelector('[style]'),
                cssModules: !!document.querySelector('[class*="__"]'),
                tailwind: !!document.querySelector('[class*="flex"], [class*="grid"], [class*="p-"]'),
                styledComponents: !!document.querySelector('[class^="sc-"]'),
                cssInJs: !!(document.querySelector('[data-css]') || document.querySelector('[data-styled]'))
            };
            
            // Zoox appears to use CSS Modules based on class naming
            if (styles.cssModules) {
                return {
                    primary: 'CSS Modules',
                    pattern: 'ComponentName_element__hash',
                    framework: 'Next.js with CSS Modules'
                };
            }
            
            return styles;
        }

        getResponsiveBreakpoints() {
            const breakpoints = [];
            const sheets = document.styleSheets;
            
            for (let sheet of sheets) {
                try {
                    const rules = sheet.cssRules || sheet.rules;
                    if (!rules) continue;
                    
                    for (let rule of rules) {
                        if (rule.media) {
                            breakpoints.push({
                                media: rule.media.mediaText,
                                rules: rule.cssRules?.length || 0
                            });
                        }
                    }
                } catch (e) {
                    continue;
                }
            }
            
            // Common breakpoints
            const commonBreakpoints = {
                mobile: breakpoints.filter(b => b.media.includes('max-width: 768') || b.media.includes('max-width: 767')),
                tablet: breakpoints.filter(b => b.media.includes('min-width: 768') && b.media.includes('max-width: 1024')),
                desktop: breakpoints.filter(b => b.media.includes('min-width: 1025'))
            };
            
            return {
                all: breakpoints,
                common: commonBreakpoints
            };
        }

        getCSSVariables() {
            const variables = [];
            const styles = window.getComputedStyle(document.documentElement);
            
            for (let i = 0; i < styles.length; i++) {
                const prop = styles[i];
                if (prop.startsWith('--')) {
                    variables.push({
                        name: prop,
                        value: styles.getPropertyValue(prop).trim()
                    });
                }
            }
            
            return variables;
        }

        getImportantSelectors() {
            const selectors = [];
            const sheets = document.styleSheets;
            
            for (let sheet of sheets) {
                try {
                    const rules = sheet.cssRules || sheet.rules;
                    if (!rules) continue;
                    
                    for (let rule of rules) {
                        if (rule.selectorText && 
                            (rule.selectorText.includes('Homepage') || 
                             rule.selectorText.includes('Content') ||
                             rule.selectorText.includes('Bespoke'))) {
                            selectors.push({
                                selector: rule.selectorText,
                                properties: rule.style?.length || 0
                            });
                        }
                    }
                } catch (e) {
                    continue;
                }
            }
            
            return selectors.slice(0, 20); // First 20 for brevity
        }

        analyzeHTMLStructure() {
            if (!this.container) return null;
            
            return {
                container: this.getElementTree(this.container),
                componentHierarchy: this.getComponentHierarchy(),
                semanticElements: this.getSemanticElements(),
                dataAttributes: this.getDataAttributes(),
                ariaAttributes: this.getAriaAttributes()
            };
        }

        getElementTree(element, depth = 0, maxDepth = 3) {
            if (depth >= maxDepth) return { name: element.tagName, truncated: true };
            
            const children = [];
            for (let child of element.children) {
                children.push(this.getElementTree(child, depth + 1, maxDepth));
            }
            
            return {
                tag: element.tagName,
                classes: element.className,
                id: element.id || null,
                children: children
            };
        }

        getComponentHierarchy() {
            const hierarchy = {};
            
            // Key Zoox components
            const componentSelectors = {
                'PageHome': '.PageHome_contentBespoke__ZkpR3',
                'HomepageContentBespoke': '.HomepageContentBespoke',
                'TextContent': '.TextContent',
                'DynamicTag': '.DynamicTag',
                'BaseImage': '.BaseImage',
                'BaseVideo': '.BaseVideo',
                'Cta': '.Cta'
            };
            
            Object.entries(componentSelectors).forEach(([name, selector]) => {
                const elements = document.querySelectorAll(selector);
                hierarchy[name] = elements.length;
            });
            
            return hierarchy;
        }

        getSemanticElements() {
            const semantic = {
                header: !!document.querySelector('header'),
                main: !!document.querySelector('main'),
                section: !!document.querySelector('section'),
                article: !!document.querySelector('article'),
                nav: !!document.querySelector('nav'),
                footer: !!document.querySelector('footer'),
                aside: !!document.querySelector('aside'),
                h1: !!document.querySelector('h1'),
                h2: !!document.querySelector('h2'),
                p: !!document.querySelector('p'),
                button: !!document.querySelector('button'),
                video: !!document.querySelector('video'),
                img: !!document.querySelector('img')
            };
            
            return semantic;
        }

        getDataAttributes() {
            const dataAttrs = new Set();
            // FIXED: Use correct selector syntax - get all elements and check their attributes
            const allElements = document.querySelectorAll('*');
            
            allElements.forEach(el => {
                Array.from(el.attributes).forEach(attr => {
                    if (attr.name.startsWith('data-')) {
                        dataAttrs.add(attr.name);
                    }
                });
            });
            
            return Array.from(dataAttrs);
        }

        getAriaAttributes() {
            const ariaAttrs = new Set();
            // FIXED: Use correct selector syntax
            const allElements = document.querySelectorAll('*');
            
            allElements.forEach(el => {
                Array.from(el.attributes).forEach(attr => {
                    if (attr.name.startsWith('aria-')) {
                        ariaAttrs.add(attr.name);
                    }
                });
            });
            
            return Array.from(ariaAttrs);
        }

        analyzeJavaScriptControls() {
            return {
                eventListeners: this.getEventListeners(),
                scrollHandlers: this.detectScrollHandlers(),
                intersectionObservers: this.detectIntersectionObservers(),
                mutationObservers: this.detectMutationObservers(),
                timers: this.detectTimers(),
                animationFrames: this.detectAnimationFrames()
            };
        }

        getEventListeners() {
            const listeners = [];
            const events = ['scroll', 'resize', 'click', 'mouseenter', 'mouseleave', 'animationend', 'transitionend'];
            
            events.forEach(event => {
                // Check for inline event handlers
                const elementsWithHandler = [];
                const allElements = document.querySelectorAll('*');
                
                allElements.forEach(el => {
                    if (el[`on${event}`]) {
                        elementsWithHandler.push(el.tagName);
                    }
                });
                
                if (elementsWithHandler.length > 0) {
                    listeners.push({
                        event,
                        count: elementsWithHandler.length
                    });
                }
            });
            
            return listeners;
        }

        detectScrollHandlers() {
            const scrollHandlers = [];
            
            // Check for scroll event listeners
            if (window.onscroll) {
                scrollHandlers.push('Window scroll handler');
            }
            
            // Check for ScrollTrigger
            if (window.ScrollTrigger) {
                scrollHandlers.push('GSAP ScrollTrigger');
            }
            
            // Check for scroll-related libraries
            if (window.LocomotiveScroll) {
                scrollHandlers.push('Locomotive Scroll');
            }
            
            // Check for scroll event in scripts
            const scripts = Array.from(document.scripts).map(s => s.textContent).join(' ');
            if (scripts.includes('addEventListener') && scripts.includes('scroll')) {
                scrollHandlers.push('Custom scroll listeners');
            }
            
            return scrollHandlers;
        }

        detectIntersectionObservers() {
            // Check if IntersectionObserver is being used
            if (window.IntersectionObserver) {
                // Look for common patterns
                const lazyImages = document.querySelectorAll('img[loading="lazy"]');
                const videoElements = document.querySelectorAll('video[autoplay]');
                
                return {
                    possible: true,
                    lazyImages: lazyImages.length,
                    videoElements: videoElements.length,
                    detected: document.scripts.length > 0 && 
                             Array.from(document.scripts).some(s => 
                                 s.textContent.includes('IntersectionObserver'))
                };
            }
            
            return false;
        }

        detectMutationObservers() {
            // Check for MutationObserver usage
            const scripts = Array.from(document.scripts).map(s => s.textContent).join(' ');
            return {
                available: !!window.MutationObserver,
                used: scripts.includes('MutationObserver')
            };
        }

        detectTimers() {
            // Check for setTimeout/setInterval usage
            const scriptContents = Array.from(document.scripts).map(s => s.textContent).join(' ');
            
            return {
                setTimeout: scriptContents.includes('setTimeout'),
                setInterval: scriptContents.includes('setInterval'),
                requestAnimationFrame: scriptContents.includes('requestAnimationFrame')
            };
        }

        detectAnimationFrames() {
            // Check for requestAnimationFrame
            return !!window.requestAnimationFrame;
        }

        detectBuildTools() {
            return {
                webpack: this.detectWebpack(),
                nextjs: !!window.__NEXT_DATA__,
                babel: this.detectBabel(),
                typescript: this.detectTypeScript()
            };
        }

        detectWebpack() {
            // Check for webpack manifest
            const scripts = document.scripts;
            for (let script of scripts) {
                if (script.src && script.src.includes('_next/static')) {
                    return {
                        present: true,
                        type: 'Next.js webpack'
                    };
                }
            }
            return false;
        }

        detectBabel() {
            // Check for Babel helpers
            const scriptContents = Array.from(document.scripts).map(s => s.textContent).join(' ');
            return scriptContents.includes('@babel') || 
                   scriptContents.includes('babel-runtime') ||
                   scriptContents.includes('regeneratorRuntime');
        }

        detectTypeScript() {
            // Check for TypeScript comments or patterns
            const scriptContents = Array.from(document.scripts).map(s => s.textContent).join(' ');
            return scriptContents.includes('/*!') && 
                   scriptContents.includes('*/') && 
                   scriptContents.includes('__extends') || 
                   scriptContents.includes('__assign');
        }

        analyzeSpecificAnimations() {
            console.log('\n%c🎯 ANIMATION BREAKDOWN', 'font-size: 14px; font-weight: bold; color: #FF5722;');
            
            // Look for specific animation patterns in Zoox
            const animatedElements = {
                horizontalText: document.querySelector('.HomepageContentBespoke_horizontalTextContainer__Jr0f0'),
                mainContainer: document.querySelector('.HomepageContentBespoke_mainContainer__cuN8J'),
                splitLines: document.querySelectorAll('.split-line'),
                freedomWords: document.querySelectorAll('.HomepageContentBespoke_words__f48IG p'),
                pinSpacer: document.querySelector('.pin-spacer'),
                clipContainer: document.querySelector('.HomepageContentBespoke_clipContainer__PaUmt')
            };
            
            Object.entries(animatedElements).forEach(([name, element]) => {
                if (element) {
                    if (name === 'splitLines' && element.length > 0) {
                        console.log(`\n${name}: ${element.length} lines with split text animation`);
                        console.log('  Animation Type: GSAP SplitText or CSS line-by-line');
                        
                        // Check first line for animation
                        const firstLine = element[0];
                        if (firstLine) {
                            const styles = window.getComputedStyle(firstLine);
                            console.log(`  Transform: ${styles.transform}`);
                            console.log(`  Opacity: ${styles.opacity}`);
                        }
                    } else if (element instanceof Element) {
                        const styles = window.getComputedStyle(element);
                        const rect = element.getBoundingClientRect();
                        
                        console.log(`\n${name}:`);
                        console.log(`  Position: ${styles.position}`);
                        console.log(`  Visible: ${rect.width > 0 ? '✅' : '❌'}`);
                        
                        if (styles.transform !== 'none') {
                            console.log(`  Transform: ${styles.transform}`);
                        }
                        if (styles.opacity !== '1') {
                            console.log(`  Opacity: ${styles.opacity}`);
                        }
                        if (element.style.translate) {
                            console.log(`  Translate: ${element.style.translate}`);
                        }
                    }
                }
            });

            // Special detection for pin-spacer (GSAP ScrollTrigger)
            if (animatedElements.pinSpacer) {
                console.log('\n📍 ScrollTrigger Pinning detected:');
                console.log('  • Main content is pinned during scroll');
                console.log('  • Height:', animatedElements.pinSpacer.style.height);
            }

            // Check for word cycling in freedom section
            if (animatedElements.freedomWords && animatedElements.freedomWords.length > 0) {
                console.log('\n🔄 Word Cycling Animation:');
                console.log('  • Words:', animatedElements.freedomWords.length);
                
                const activeWord = document.querySelector('.HomepageContentBespoke_active__mJOsI');
                if (activeWord) {
                    console.log(`  • Currently active: "${activeWord.textContent.trim()}"`);
                }
            }

            // Check for clip path animation
            if (animatedElements.clipContainer) {
                const styles = window.getComputedStyle(animatedElements.clipContainer);
                console.log('\n✂️ Clip Path Animation:');
                console.log(`  • Clip Path: ${styles.clipPath || styles.webkitClipPath || 'None'}`);
            }
        }

        getAnimationTimeline() {
            console.log('\n%c⏱️ ANIMATION TIMELINE', 'font-size: 14px; font-weight: bold; color: #FFC107;');
            
            console.log('Scroll-based animations:');
            
            // Detect scroll-based animations
            const pinSpacer = document.querySelector('.pin-spacer');
            if (pinSpacer) {
                console.log('  📍 Section 1: Pinned main container');
                console.log('     • Triggers when scrolled into view');
                console.log('     • Content fixed while pin-spacer takes space');
            }
            
            // Check for horizontal scroll
            const horizontalText = document.querySelector('.HomepageContentBespoke_horizontalTextContainer__Jr0f0');
            if (horizontalText) {
                const transform = window.getComputedStyle(horizontalText).transform;
                console.log('  ↔️ Section 2: Horizontal text parallax');
                console.log('     • Text moves on scroll: ' + (transform !== 'none' ? transform : 'active'));
            }
            
            // Check for word cycling
            const freedomWords = document.querySelectorAll('.HomepageContentBespoke_words__f48IG p');
            if (freedomWords.length > 1) {
                console.log('  🔄 Section 3: Word cycling carousel');
                console.log('     • Rotates through: Workspace, Chill Space, Photo Booth, Date Night, Nightclub');
                console.log('     • Updates every few seconds with fade transition');
            }

            // Check for video
            const video = document.querySelector('video');
            if (video) {
                console.log('  🎬 Section 4: Background video');
                console.log('     • Autoplaying looped video');
                console.log('     • Custom controls with progress tracking');
            }

            console.log('\nAnimation triggers:');
            console.log('  • Scroll position');
            console.log('  • Viewport entry (Intersection Observer)');
            console.log('  • Timer-based (word cycling)');
            console.log('  • User interaction (CTA buttons)');
        }

        generateReport() {
            console.log('%c🏗️ ZOOX PAGE ARCHITECTURE ANALYSIS', 'font-size: 18px; font-weight: bold; color: #FF4081;');
            console.log('%c=====================================', 'font-weight: bold;');
            
            // Framework
            console.log('\n%c🔧 FRAMEWORK', 'font-size: 14px; font-weight: bold; color: #2196F3;');
            console.log('Framework:', this.framework.name || 'React/Next.js detected');
            if (this.framework.version) console.log('Version:', this.framework.version);
            
            // Animations
            console.log('\n%c🎬 ANIMATION LIBRARY', 'font-size: 14px; font-weight: bold; color: #4CAF50;');
            if (this.animationLibrary.library === 'GSAP') {
                console.log('Library:', this.animationLibrary.library);
                console.log('Version:', this.animationLibrary.version);
                console.log('Plugins:', this.animationLibrary.plugins.join(', ') || 'None detected');
                console.log('Animated Elements:', this.animationLibrary.instances.length);
            } else {
                console.log('CSS Animations:', this.detectCSSAnimations().length);
            }
            
            // CSS Architecture
            console.log('\n%c🎨 CSS ARCHITECTURE', 'font-size: 14px; font-weight: bold; color: #FF9800;');
            console.log('Methodology:', this.cssArchitecture.methodology);
            console.log('Styling Approach:', this.cssArchitecture.stylingApproach.primary || 'CSS Modules');
            console.log('Components Found:', Object.keys(this.cssArchitecture.components).length);
            console.log('CSS Variables:', this.cssArchitecture.cssVariables.length);
            
            // Responsive Breakpoints
            console.log('\n%c📱 RESPONSIVE BREAKPOINTS', 'font-size: 14px; font-weight: bold; color: #9C27B0;');
            console.log('Mobile:', this.cssArchitecture.responsiveBreakpoints.common.mobile.length, 'queries');
            console.log('Tablet:', this.cssArchitecture.responsiveBreakpoints.common.tablet.length, 'queries');
            console.log('Desktop:', this.cssArchitecture.responsiveBreakpoints.common.desktop.length, 'queries');
            
            // HTML Structure
            console.log('\n%c📄 HTML STRUCTURE', 'font-size: 14px; font-weight: bold; color: #E91E63;');
            console.log('Semantic Elements:', Object.values(this.htmlStructure.semanticElements).filter(Boolean).length);
            console.log('Data Attributes:', this.htmlStructure.dataAttributes.length);
            console.log('ARIA Attributes:', this.htmlStructure.ariaAttributes.length);
            
            // JavaScript Controls
            console.log('\n%c⚡ JAVASCRIPT CONTROLS', 'font-size: 14px; font-weight: bold; color: #00BCD4;');
            console.log('Scroll Handlers:', this.javascriptControls.scrollHandlers.join(', ') || 'None detected');
            console.log('Event Listeners:', this.javascriptControls.eventListeners.length, 'types');
            console.log('Intersection Observer:', this.javascriptControls.intersectionObservers?.possible ? '✅ Used' : '❌ Not detected');
            
            // Build Tools
            console.log('\n%c🛠️ BUILD TOOLS', 'font-size: 14px; font-weight: bold; color: #795548;');
            console.log('Next.js:', this.buildTools.nextjs ? '✅' : '❌');
            console.log('Webpack:', this.buildTools.webpack ? '✅' : '❌');
            console.log('TypeScript:', this.buildTools.typescript ? '✅' : '❌');
            
            // Animation Analysis
            this.analyzeSpecificAnimations();
            this.getAnimationTimeline();
            
            return {
                framework: this.framework,
                animation: this.animationLibrary,
                css: this.cssArchitecture,
                html: this.htmlStructure,
                js: this.javascriptControls,
                build: this.buildTools
            };
        }
    }

    // Initialize and run analysis
    window.zooxArchitecture = new ZooxArchitectureAnalyzer();
    
    console.log('%c✅ Zoox Architecture Analyzer Loaded!', 'font-size: 14px; font-weight: bold; color: #4CAF50;');
    console.log('\n%c📋 Available Commands:', 'font-size: 14px; font-weight: bold;');
    console.log('1. %czooxArchitecture.generateReport()', 'color: #2196F3;', '- Complete architecture analysis');
    console.log('2. %czooxArchitecture.getAnimationTimeline()', 'color: #2196F3;', '- Animation breakdown');
    console.log('3. %czooxArchitecture.analyzeSpecificAnimations()', 'color: #2196F3;', '- Detailed animation analysis');
    console.log('4. %cJSON.stringify(zooxArchitecture, null, 2)', 'color: #2196F3;', '- Export all data');

    return window.zooxArchitecture;
})();
